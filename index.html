Código Python para Estudo Colégio Dinâmica

print("Olá, Mundo!")
produto = []
def menuProduto():
    while True:
        print("Menu Produto")
        print("1- Cadastro Produto")
        print("2- Mostrar Produto")
        print("3- Sair")
        opcao = input("Escolha uma opção: ")
        if opcao == '1':
            CadProduto()
        elif opcao == '2':
            print(produto)
        elif opcao == '3':
            break
        else: 
            print("Usuário querido, não viu as opções acima!")
def CadProduto():
    id = int(input("Digite o ID: "))
    nome = input("Digite o Produto: ")
    marca = input("Digite a Marca: ")
    modelo = input("Digite o Modelo: ")
    custo = float(input("Digite o Custo: "))
    produto.append([id, nome, marca, modelo, custo])
def mostrarProd():
    if not produto:
        print("Nenhum Produto cadastrado.")
    else:
        print("\nProdutos no Inventário:")
        for i, prod in enumerate(produto):
            print(f"Produto {i+1}: ID:{prod[0]} - Nome:{prod[1]}  "
                  f"- Modelo:{prod[2]} - Marca:{prod[3]} - "
                  f"Custo:R${prod[4]:,.2f} - Estoque:{prod[5]} - "
                  f"Lucro:R${prod[6]:,.2f}")
menuProduto()
